   * @param json
   * @Description 性能数据写入文件
   * @Author Liuws
   * @Date 2021/9/15 17:56
   */
  public String writePerfData(Long resourceId, String metricName, JSONArray json) throws Exception {
    //性能数据写入文件路径
    String filePath =
        PathUtil.getLogPath() + "/logs/alarm/powerAttenuation/perfData" + File.separator + metricName + File.separator + resourceId + "_" + metricName + ".txt";
    File file = new File(filePath);
    // 创建文件存放的目录
    if (!file.getParentFile().exists()) {
      file.getParentFile().mkdirs();
    }
    try {
      FileOutputStream fileOutputStream = new FileOutputStream(file);
      BufferedWriter bufferedWriter = new BufferedWriter(new OutputStreamWriter(fileOutputStream));
      bufferedWriter.write(json.toString());
      bufferedWriter.close();
      fileOutputStream.close();
      return filePath;
    } catch (Exception e) {
      logger.error("性能数据写入文件失败:", e.getMessage());
      e.printStackTrace();
      throw new Exception(e);
    }
  }
